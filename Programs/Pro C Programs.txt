#include<stdio.h>
#include<conio.h>
#include<string.h>

exec sql include sqlca;
exec sql begin declare section;
char uid[250];
int id;
char name[20];
float salary;
exec sql end declare section;

void err_report() {
	printf("Error Generated...");
	printf("Error msg %s \n ",sqlca.sqlerrm.sqlerrmc);
	exit(0);
}



void main() {
	 strcpy(uid,"scott/tiger@localhost:1521/orcl");
	 EXEC SQL WHENEVER SQLERROR DO err_report();
	 printf("Enter the id \n");
	 scanf("%d",&id);
	 printf("Enter the name\n");
	 scanf("%s",&name);
	 printf("Enter the salary\n");
	 scanf("%f",&salary);
     exec sql connect :uid;
     EXEC SQL insert into employee values(:id,:name);
	 EXEC SQL COMMIT WORK;
	 printf("Record inserted successfuly....");
	 getch();
}
